数据结构与算法-栈

发表于 2017-09-03

栈是哟中遵从先进后出原则的有序集合。新添加的或待删除的元素都保存在栈的末尾，称做栈顶另一端叫栈底。在栈里新元素都靠近栈顶，旧元素都靠进栈底。
栈的创建
`

function Stack(){

// 各种属性和方法的声明
var items = [];
// 在栈顶添加元素
this.push = function(element){
items.push(element);
};
// 删除栈顶元素，同时返回被删除的元素
this.pop = function(){
return items.pop();
};
// 返回栈顶元素，不做任何修改
this.peek = function(){
return items[items.length - 1];
};
// 当栈里没有任何元素时返回true 否则返回false
this.isEmpty = function(){
return items.length == 0;
};
// 返回栈里元素的个数
this.size = function(){
return items.length;
};
// 清除栈里面的所有元素
this.clear = function(){
items = [];
};
// 将栈里的所有元素输出到控制台
this.print = function(){
console.log(items.toString());
}
// 将十进制转换为二进制
function divideBy2(decNumber,base){
//base 将要转换的进制2，8，10，16
var remStack = new Stack(),
rem,
binaryString = ‘’,
digits = ‘0123456789ABCDEF’;
while(decNumber > 0){
rem = Math.floor(decNumber % base);
remStack.push(rem);
decNumber = Math.floor(decNumber / base);
}
while(!remStack.isEmpty()){
binaryString += digits[remStack.pop()];
}
return binaryString;
}
console.log(divideBy2(100036984562,8));
`
