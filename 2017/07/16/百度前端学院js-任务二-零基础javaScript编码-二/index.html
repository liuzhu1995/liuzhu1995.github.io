百度前端学院js 任务二:零基础javaScript编码(二)

发表于 2017-07-16 
**任务描述**
参考以下示例代码，页面加载后，将提供的空气质量数据数组，按照某种逻辑（比如空气质量大于60）进行过滤筛选，最后将符合条件的数据按    照一定的格式要求显示在网页上
`
<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <title>Title</title>
</head>
<body>
    <h3>污染城市列表</h3>
    <ul id="api-list">
        <!--
    <li>第一名：福州（样例），10</li>
      <li>第二名：福州（样例），10</li> -->
    </ul>
    <script type="text/javascript">

        var aqiData = [
            ["北京", 90],
            ["上海", 50],
            ["福州", 10],
            ["广州", 50],
            ["成都", 90],
            ["西安", 100]
        ];

        (function () {

            /*
             在注释下方编写代码
             遍历读取aqiData中各个城市的数据
             将空气质量指数大于60的城市显示到aqi-list的列表中
             */
            //获取ul元素
            var api_list = document.getElementById("api-list");
            //要获得名次
            var ranking = ['一','二','三','四','五','六'];
            //用sort()方法对城市进行排名
           /* aqiData.sort(compare);
            function compare(a,b){
                return b[1] - a[1]
            }*/
           aqiData.sort(function(a,b){
               return b[1] - a[1]; // [1]利用数组的索引值获取数组的数字进行排序
           })
            //遍历所有数组
            for(var i = 0;i<aqiData.length;i++){
               //判断数字是否大于60
                if(aqiData[i][1] > 60){
                    //创建li元素
                    var lis = document.createElement('li');
                    //设置创建的li元素的文本内容
                    lis.innerHTML = '第' + ranking[i] + '名:' + aqiData[i][0] +','+ aqiData[i][1];
                    //将li节点添加到ul节点中
                    api_list.appendChild(lis);
                }
            }
        })();

    </script>
</body>
</html>
`
