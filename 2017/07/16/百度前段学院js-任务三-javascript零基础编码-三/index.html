**任务描述**
参考以下示例代码，读取页面上已有的source列表，从中提取出城市以及对应的空气质量
将数据按照某种顺序排序后，在resort列表中按照顺序显示出来
`
<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <title>Title</title>
</head>
<body>
<!--参考以下示例代码，读取页面上已有的source列表，从中提取出城市以及对应的空气质量
将数据按照某种顺序排序后，在resort列表中按照顺序显示出来-->
<ul id="source">
    <li>北京空气质量：<b>90</b></li>
    <li>上海空气质量：<b>70</b></li>
    <li>天津空气质量：<b>80</b></li>
    <li>广州空气质量：<b>50</b></li>
    <li>深圳空气质量：<b>40</b></li>
    <li>福州空气质量：<b>32</b></li>
    <li>成都空气质量：<b>90</b></li>
</ul>

<ul id="resort">
    <!--
    <li>第一名：北京空气质量：<b>90</b></li>
    <li>第二名：北京空气质量：<b>90</b></li>
    <li>第三名：北京空气质量：<b>90</b></li>
     -->

</ul>

<button id="sort-btn">排序</button>

<script type="text/javascript">
    (function(){
        function getData(){
            //        定义一个空数组
            var data = [];
//        获取ul节点的子节点集合
            var lis = document.getElementById('source').childNodes;
//        遍历它们
            for(var i=0;i<lis.length;i++){
                if(lis[i].innerHTML != undefined){
//                获取城市名称  substring()用于提取字符串中介于两个指定下标之间的字符
                    var city = lis[i].innerHTML.substring(0,2);
//                获取第二个子节点的文本内容也就是数字
                    var num = lis[i].childNodes[1].innerHTML;
//                将城市名称和数字添加进入数组
                    data.push([city,num]);
                }
            }
            return data;
        }
//排序
        function sortArray(data) {
            return data.sort(function (a, b) {
                return b[1] - a[1];
            })
        }
            //添加到文本当中
        function addEle(data){
            var ranking = ['一','二','三','四','五','六','七'];
//            遍历
            for(var i = 0;i<data.length;i++){
                //创建li元素
                var li = document.createElement('li');
                //添加内容
                li.innerHTML = '第' + ranking[i] + '名:' + data[i][0] + '空气质量:';
                //创建b元素
                var b = document.createElement('b');
                //为b元素添加文本内容
                b.innerHTML = data[i][1];
                //向li节点添加最后一个子节点b
                li.appendChild(b);
                //获取ul元素
                var ul = document.getElementById('resort');
                //向ul节点添加最后一个子节点li
                ul.appendChild(li);
            }

        }
        function a(){
            var aqiData = getData();
            aqiData = sortArray(aqiData);
            addEle(aqiData);
        }
//        添加点击事件
            var btn = document.getElementById('sort-btn');
            btn.onclick = function(){
               a()
            }


    })()


</script>
</body>
</html>
`
