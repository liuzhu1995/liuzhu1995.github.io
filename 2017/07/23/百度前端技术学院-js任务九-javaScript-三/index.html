**任务描述**
基于任务九，添加节点的选择、增加与删除的功能
点击某个节点元素，则该节点元素呈现一个特殊被选中的样式
增加一个删除按钮，当选中某个节点元素后，点击删除按钮，则将该节点及其所有子节点删除掉
增加一个输入框及一个“添加”按钮当选中某个节点元素后，点击增加按钮，则在该节点下增加一个子节点，节点内容为输入框中内容，插入在其子节点的最后一个位置

之前有好多代码虽然功能实现了，但是比较繁琐，这次已经改进了不少。

    <style>
        #root{
            position: relative;

            left: 50%;

        div{
            display: flex;

        }
        .btn{
            margin: 20px auto;

        }
        input[type=button]{
            display: inline-block;
            color: #fff;
            font-size: 15px;
            font-weight:bolder;
            background: rgba(0,0,0,0.5);
            padding: 8px 5px;
            border:none;
            cursor:pointer;
        }
        input[type=text]{
            padding: 8px 5px;
            border: 2px solid rgba(0,0,0,0.5);
        }
        input[type=button]:hover{
            background: rgba(0,0,0,0.8);
        }
        .select{
            background: pink;
        }
    </style>
</head>
<body>
    <div id="root">想笑来伪装掉下的眼泪
        <div>点点头承认自己会怕黑

        <div>天灰灰会不会让我忘了你是谁
            <div>夜越黑梦违背难追难回味</div>
            <div>我的世界将被摧毁也许事与愿违
                <div>累不累睡不睡单影无人相依偎</div>
                <div>夜越黑梦违背有谁肯安慰</div>
            </div>
        </div>　也许颓废也是……
    </div>
    <div class="btn">
        <input type="button" value="前序遍历" id="btn_left">
        <input type="button" value="后序遍历" id="btn_right">
        <input type="text" id="val">
        <input type="button" value="搜索" id="inquiry">
        <input type="text" id="content">
        <input type="button" value="添加" id="add">
        <input type="button" value="删除" id="remove">
    </div>
    <script>
        var btn_left = document.getElementById("btn_left"),
            btn_right = document.getElementById("btn_right"),
            inquiry = document.getElementById("inquiry"),
            add = document.getElementById("add"),
            remove = document.getElementById("remove"),
            root  = document.getElementById("root"),
            arr = [],
            boole = false,
            previous,
            time;
 //每次点击前调用复原
        function reset(){
            var divs = document.getElementsByTagName("div");
            for(var i=0;i<divs.length;i++){
                divs[i].classList.remove("select");
            }
            arr=[];
            if (previous) {
                previous.style.background = "white";
            }
        }


        //前序遍历
        function preOrder(node){
            if(!(node==null)){
                arr.push(node);
                preOrder(node.firstElementChild);
                preOrder(node.lastElementChild);
            }
        }
        //后序遍历
        function postOrder(node){
            if(!(node == null)){
                postOrder(node.firstElementChild);
                postOrder(node.lastElementChild);
                arr.push(node);
            }
        }
    //显示遍历过程
        function addColor(){
            for(var i=0;i<arr.length;i++){
                (function(i){
                    setTimeout(function(){
                        if(i == arr.length-1){
                            boole = false;
                        }
                        if(previous){
                            previous.style.backgroundColor = "white";
                        }
                        arr[i].style.backgroundColor = "greenyellow";
                        previous = arr[i];
                    },i*200)
                })(i)
            }
        }

        //查询过程函数函数
        function quiry(){
            var i=0,
                val = document.getElementById("val");
            if(val.value == ""){
                alert("请输入查询内容");
                boole = false;
            }else{
                time = setInterval(function(){
                    if(i<arr.length){
                        if(i-1 >-1){
                            arr[i-1].style.backgroundColor = "white";
                        }
                        arr[i].style.backgroundColor = "greenyellow";
                        i++
                    }else{
                        boole = false;
                        arr[i-1].style.backgroundColor = "white";
                        clearInterval(time);
                        show();
                    }
                },200)
            }

        }
        //查询函数将结果显示
        function show(){
            var light = [];
            var val = document.getElementById("val").value;
            var regexp = eval("/"+ val +"/g");
            for(var i=0;i<arr.length;i++){
                var text = arr[i].innerText;
               if(regexp.test(text)){
                   light.push(i)
               }
            }
            if(light.length>0){
                alert("已匹配到" +light.length+"个");
                for(var j=0;j<light.length;j++){
                    var x = light[j];
                    arr[x].style.backgroundColor = "greenyellow";
                }
            }else{
                alert("未匹配到相似元素")
            }
        }

        //添加类名

        btn_left.onclick = function(){
            if(!boole) {
                boole = true;
                    reset();
                    preOrder(root);
                    addColor()
            }

        };
        btn_right.onclick = function(){
            if(!boole) {
                boole = true;
                reset();
                postOrder(root);
                addColor()
            }
        };
//查询
        inquiry.onclick = function(){
            clearInterval(time);
            quiry();
        };

        //点击添加类名
        (function(){
            preOrder(root);
           for(var i=0;i<arr.length;i++){
               arr[i].onclick = function(e){
                   e.stopPropagation();//防止事件冒泡;
                   for(var i=0;i<arr.length;i++){
                       arr[i].classList.remove("select")
                   }
                   this.classList.add("select");
               }
           }
        })();

        //删除
        function del(){
            preOrder(root);
            arr.forEach(function(e){
                var name = e.className;
                if(name == "select"){
                  e.remove(this);
                }
            })
        }
        //调用删除函数
        remove.onclick = function(){
            del();
        }

        //添加
        function append(){
            var num = [];
            var divs = document.getElementsByTagName("div");
            for(var i=0;i<divs.length;i++){
                if(divs[i].className == "select"){
                    num.push(i);
                }
            }
            var content = document.getElementById("content").value;
            if(content == ""){
                alert("请输入要添加的内容")
            }else if(num == 0){
                alert("请选择要指定添加内容的区域")
            } else{
                var div = document.createElement("div");
                div.innerHTML = content;
                var select = document.getElementsByClassName("select")[0];
                select.appendChild(div);
            }

        }
        add.onclick = function(){
            append();
        }
    </script>
</body>
</html>
